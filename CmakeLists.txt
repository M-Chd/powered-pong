cmake_minimum_required(VERSION 3.10)
project(PoweredPong)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(PoweredPong
    src/main.cpp
    src/entities/ball.cpp
    src/entities/player.cpp
    src/core/window.cpp
    
    src/core/game.cpp
  "src/core/board.h" "src/core/board.cpp" "src/entities/item.cpp" "src/core/gameConfig.h" 
  "src/core/gameConfig.cpp" "src/util/vector2.cpp" "src/core/inputManager.h" 
  "src/core/inputManager.cpp" "src/ui/view.h" "src/ui/scoreBoard.cpp" "src/core/textureCache.h" "src/util/util.h" "src/ui/debugOverlay.h" "src/ui/debugOverlay.cpp")

target_include_directories(PoweredPong PRIVATE
    src
    src/util
    src/entities
    src/core
    src/ui
)

find_package(SDL2 CONFIG REQUIRED)
find_package(SDL2_ttf CONFIG REQUIRED)
find_package(SDL2_image CONFIG REQUIRED)

target_link_libraries(PoweredPong PRIVATE SDL2::SDL2 SDL2::SDL2main SDL2_ttf::SDL2_ttf SDL2_image::SDL2_image)
