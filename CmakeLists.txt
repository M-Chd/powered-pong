cmake_minimum_required(VERSION 3.10)
project(PoweredPong)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(PoweredPong
    src/main.cpp
    src/entities/ball.cpp
    src/entities/player.cpp
    src/core/window.cpp
    src/core/view.cpp
    src/core/game.cpp
  "src/core/board.h" "src/core/board.cpp" "src/entities/item.cpp" "src/core/gameConfig.h" "src/core/gameConfig.cpp" "src/util/vector2.cpp")

target_include_directories(PoweredPong PRIVATE
    src
    src/util
    src/entities
    src/core
)

find_package(SDL2 CONFIG REQUIRED)

target_link_libraries(PoweredPong PRIVATE SDL2::SDL2 SDL2::SDL2main)
